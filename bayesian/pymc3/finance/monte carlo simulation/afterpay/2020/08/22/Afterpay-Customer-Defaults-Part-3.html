Robustness Study

We now have ability to compute the expected average late fee, given a distribution of transaction values, and a probability of making payment on time.

Given this, we can try lots of different distributions, and use this to find the distribution of average late fees that we could see.

def create_late_fee_distribution():
    unique, counts = np.unique(samples["total_fees"], return_counts=True)

    d = {}
    for i in range(len(unique)):
        d[i] = unique[i]


    with pm.Model() as catagorical_fee_model:

        late_fee_distribution_aud[samples["late_fee_distribution_catagorical"] == i] = unique[i]

import random

def compute_late_fee(transaction_value,probability_of_late_payment = 0.5):
    total_fees = 0
    
    number_of_delays_of_less_than_1_week = 1 + np.random.binomial(3,probability_of_late_payment,1)[0]
    
    if number_of_delays_of_less_than_1_week > 0:
        number_of_delays_of_more_than_1_week = np.random.binomial(number_of_delays_of_less_than_1_week, probability_of_late_payment,1)[0]
    
    total_fees = 10 * number_of_delays_of_less_than_1_week + 7 * number_of_delays_of_more_than_1_week
    late_fee = min([total_fees,0.25*transaction_value])
    late_fee = max([late_fee,10])
    return(late_fee)
